class publisher():
    def __init__(self,pname):
        self.pname=pname
    def display(self):
        print("name:",self.pname)
class book(publisher):
    def __init__(self,pname,bname,title):
        self.pname=pname
        self.bname=bname
        self.title=title
    def display(self):
        print("name:",self.pname)
        print("book:",self.bname)
        print("title:",self.title)
